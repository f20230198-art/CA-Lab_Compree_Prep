    Quick Reference¶

    üî¢ Syscall Numbers¶

    Syscall Code Arguments Returns
    Print Integer li $v0, 1 $a0 = integer -
    Print Float li $v0, 2 $f12 = float -
    Print Double li $v0, 3 $f12 = double -
    Print String li $v0, 4 $a0 = address -
    Read Integer li $v0, 5 - $v0 = integer
    Read Float li $v0, 6 - $f0 = float
    Read Double li $v0, 7 - $f0 = double
    Read String li $v0, 8 $a0 = buffer, $a1 = length -
    Allocate Heap li $v0, 9 $a0 = bytes $v0 = address
    Exit Program li $v0, 10 - -

    üìù Register Conventions¶

    Temporary Registers ($t0-$t9)¶

    • Use: Temporary calculations
    • Saved: NO (caller saves if needed)
    • Example: Loop counters, intermediate values

    Saved Registers ($s0-$s7)¶

    • Use: Important values across function calls
    • Saved: YES (callee saves)
    • Example: Variables that persist

    Argument Registers ($a0-$a3)¶

    • Use: Function arguments
    • Example: Parameters to procedures

    Return Value Registers ($v0-$v1)¶

    • Use: Function return values
    • Example: Results from procedures

    Special Registers¶

    • $zero (always 0)
    • $sp (stack pointer)
    • $ra (return address)

    üîÑ Common Instructions¶

    Arithmetic¶

    add  $t0, $t1, $t2    # $t0 = $t1 + $t2
sub  $t0, $t1, $t2    # $t0 = $t1 - $t2
mul  $t0, $t1, $t2    # $t0 = $t1 √ó $t2
div  $t1, $t2         # LO = $t1 √∑ $t2, HI = $t1 % $t2
mflo $t0              # $t0 = LO
mfhi $t0              # $t0 = HI

addi $t0, $t1, 5      # $t0 = $t1 + 5

    Memory Operations¶

    lw   $t0, 0($t1)      # $t0 = Memory[$t1]
sw   $t0, 0($t1)      # Memory[$t1] = $t0
lb   $t0, 0($t1)      # Load byte
sb   $t0, 0($t1)      # Store byte
la   $t0, label       # $t0 = address of label
li   $t0, 100         # $t0 = 100

    Branching¶

    beq  $t0, $t1, label  # if ($t0 == $t1) goto label
bne  $t0, $t1, label  # if ($t0 != $t1) goto label
blt  $t0, $t1, label  # if ($t0 < $t1) goto label
ble  $t0, $t1, label  # if ($t0 <= $t1) goto label
bgt  $t0, $t1, label  # if ($t0 > $t1) goto label
bge  $t0, $t1, label  # if ($t0 >= $t1) goto label
j    label            # goto label
jr   $ra              # return (jump to $ra)
jal  label            # call function (saves PC in $ra)

    üìã Common Code Patterns¶

    1. Basic Program Structure¶

    .data
    # Variables here
    msg: .asciiz "Hello"
    num: .word 10

.text
.globl main
main:
    # Your code here

    # Exit
    li $v0, 10
    syscall

    2. Print Integer¶

    li $v0, 1        # syscall code for print int
la $a0, num      # load value to print
syscall          # print

    3. Print String¶

    li $v0, 4        # syscall code for print string
la $a0, msg      # load address of string
syscall          # print

    4. Read Integer¶

    li $v0, 5        # syscall code for read int
syscall          # read
move $t0, $v0    # save to $t0

    5. For Loop (i = 0; i < n; i++)¶

        li $t0, 0        # i = 0
    li $t1, 10       # n = 10
loop:
    bge $t0, $t1, end_loop  # if i >= n, exit

    # Loop body here

    addi $t0, $t0, 1  # i++
    j loop
end_loop:

    6. While Loop (while condition)¶

    while:
    # Check condition
    beq $t0, $zero, end_while  # if condition false, exit

    # Loop body

    j while
end_while:

    7. If-Else¶

        bne $t0, $t1, else  # if $t0 != $t1, go to else
    # if body
    j end_if
else:
    # else body
end_if:

    8. Array Access (array[i])¶

        la $t0, array    # base address
    li $t1, 2        # index i

    # Calculate offset: i * 4 (word size)
    sll $t2, $t1, 2  # $t2 = i * 4
    add $t3, $t0, $t2  # $t3 = array + offset

    lw $t4, 0($t3)   # $t4 = array[i]

    9. Simple Procedure¶

    # Caller
    li $a0, 5         # argument
    jal function      # call
    move $t0, $v0     # result in $v0

# Procedure
function:
    # Use $a0 as input
    # Process
    move $v0, $a0     # set return value
    jr $ra            # return

    10. String Length¶

        la $t0, string   # address
    li $t1, 0        # counter
length_loop:
    lb $t2, 0($t0)   # load byte
    beq $t2, $zero, length_done  # if null terminator
    addi $t1, $t1, 1  # count++
    addi $t0, $t0, 1  # next char
    j length_loop
length_done:
    # $t1 has length

    üéØ Common Mistakes to Avoid¶

    ‚ùå Forgetting to exit 

    # Wrong - program crashes
main:
    # code
    # missing exit!

# Right
main:
    # code
    li $v0, 10
    syscall

    ‚ùå Wrong array indexing 

    # Wrong - forgot to multiply by 4
add $t0, $array, $index

# Right - multiply by word size
sll $t1, $index, 2
add $t0, $array, $t1

    ‚ùå Not saving registers in procedures 

    # Wrong - overwrites caller's $t0
function:
    li $t0, 5
    jr $ra

# Right - use stack or $v registers
function:
    li $v0, 5
    jr $ra

    ‚ùå Infinite loops 

    # Wrong - never increments
loop:
    # body
    j loop

# Right - has increment
loop:
    # body
    addi $t0, $t0, 1
    blt $t0, $t1, loop

    üí° Exam Day Tips¶

    1. Write structure first 

      .data
.text
.globl main
main:
    li $v0, 10
    syscall

    2. Use meaningful labels

    3. loop, end_loop, if_true, else, etc.

    4. Comment your code

    5. Helps you and gives partial credit

    6. Test with small values mentally

    7. Walk through with example inputs

    8. Check common errors

    9. Exit syscall?
    10. Array indexing * 4?
    11. Null terminator for strings?
